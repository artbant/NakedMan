## Финишная точка — Портал
Рисуется как пиксельная арка в конце уровня (тайл 57-58, ряд 5-7):

```js
// portal object:
const portal = { x: 57 * T, y: 5 * T, w: T, h: T * 2, active: true };

function drawPortal() {
  if (!portal.active) return;
  const px = Math.round(portal.x - cam);
  const py = portal.y;

  // мигающая рамка
  const blink = Math.floor(Date.now() / 200) % 2;
  ctx.fillStyle = blink ? '#E4E4E4' : '#888';

  // арка попиксельно
  ctx.fillRect(px + 4, py, T - 8, 2);      // верх
  ctx.fillRect(px, py + 2, 4, T * 2 - 4); // левая стойка
  ctx.fillRect(px + T - 4, py + 2, 4, T * 2 - 4); // правая стойка
  ctx.fillRect(px + 4, py + T * 2 - 2, T - 8, 2); // низ

  // внутри — мигающие пиксели
  ctx.fillStyle = blink ? '#555' : '#333';
  for (let dy = 4; dy < T * 2 - 4; dy += 4)
    for (let dx = 6; dx < T - 6; dx += 4)
      ctx.fillRect(px + dx, py + dy, 2, 2);
}

function checkPortal() {
  const p = player;
  if (p.x < portal.x + portal.w && p.x + p.w > portal.x &&
      p.y < portal.y + portal.h && p.y + p.h > portal.y) {
    // ПОБЕДА
    showLevelComplete();
  }
}
```

## Экран победы:
```js
let levelComplete = false;

function showLevelComplete() {
  levelComplete = true;
}

function drawLevelComplete() {
  if (!levelComplete) return;
  ctx.fillStyle = 'rgba(0,0,0,0.88)';
  ctx.fillRect(0, 0, GW, GH);
  pixelText('YOU MADE IT', GW / 2 - 38, GH / 2 - 14, '#E4E4E4', 2);
  pixelText('PRESS JUMP', GW / 2 - 30, GH / 2 + 14, '#888', 1);
}
```
