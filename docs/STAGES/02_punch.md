## Частицы при ударе
```js
// массив particles = []
// при ударе (ptimer === 0):
function spawnHitParticles(x, y) {
  for (let i = 0; i < 12; i++) {
    const ang = Math.random() * Math.PI * 2;
    const spd = 20 + Math.random() * 50;
    particles.push({
      x, y,
      vx: Math.cos(ang) * spd,
      vy: Math.sin(ang) * spd - 20,
      life: 1,
      maxLife: 0.3 + Math.random() * 0.2,
      size: 1,
    });
  }
}

// в drawParticles():
function drawParticles() {
  for (const p of particles) {
    ctx.fillStyle = `rgba(228,228,228,${p.life.toFixed(2)})`;
    ctx.fillRect(Math.round(p.x), Math.round(p.y), p.size, p.size);
  }
}
```

## Логика в update():
```js
const isPunch = keys['KeyZ'] || keys['KeyX'] || mk.punch;

if (p.pcooldown > 0) p.pcooldown -= dt;

if (isPunch && !p.punching && p.pcooldown <= 0 && p.onGround) {
  p.punching = true;
  p.ptimer = 0;
  p.pframe = 0;
}

if (p.punching) {
  p.ptimer += dt;
  if (p.ptimer < 0.08) p.pframe = 0;
  else if (p.ptimer < 0.18) p.pframe = 1;
  else if (p.ptimer < 0.28) p.pframe = 2;
  else { p.punching = false; p.pcooldown = 0.15; }

  // спаун частиц при pframe 2
  if (p.pframe === 2 && p.ptimer < 0.19) {
    const hx = p.facing > 0 ? Math.round(p.x - cam) + p.w + 4 : Math.round(p.x - cam) - 4;
    spawnHitParticles(hx, Math.round(p.y) + 12);
  }

  // во время удара не двигаемся
  p.vx = 0;
}
```
